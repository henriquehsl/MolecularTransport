# Molecular Dynamics Poiseuille Flow Simulation Parameters
# This file shows the default parameters used in the simulation

# System parameters
nparticles = 500        # Number of particles
box_x = 10.0           # Box size in x direction
box_y = 10.0           # Box size in y direction (between walls)
box_z = 20.0           # Box size in z direction (flow direction)

# Simulation parameters
dt = 0.005             # Time step
nsteps = 10000         # Number of simulation steps
output_freq = 100      # Output frequency
temperature = 1.0      # System temperature

# Force field parameters
sigma = 1.0            # Lennard-Jones sigma parameter
epsilon = 1.0          # Lennard-Jones epsilon parameter
cutoff = 2.5           # LJ interaction cutoff distance

# Flow parameters
wall_force = 0.1       # External force driving the flow
density = 0.8          # Particle density

# Analysis parameters
profile_bins = 50      # Number of bins for velocity profile analysis

# Notes:
# - The simulation uses reduced units where mass = 1, sigma = 1, epsilon = 1
# - Poiseuille flow is generated by applying a constant force in z-direction
# - Walls are at y = 0 and y = box_y with reflecting boundary conditions
# - Periodic boundary conditions are applied in x and z directions
# - Viscosity is estimated from the velocity gradient in the flow profile
